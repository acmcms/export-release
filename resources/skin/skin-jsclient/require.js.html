require.js

for script to be able to access some classes, other scripts, report data, etc. from the server.

// Usage like this: // // require( "[<namespace>,...]" ); // example: var Layout = require("Layouts.Layout"); // example: var Coordinates = require("Utils.Coordinates"); // // -------------------------------------------------------------------- // // requireScript( <javaScriptFileName>[, dynamic ] ); // Returns script execution result. // Script will be executed only once in every window where this method is called. // Script sources will be searched in parent windows and cached in current window. // Dynamic scripts are not cached ever, use them for data access, be sure not to // initialize global variables in you scripts. // example: var Layout = requireScript("Layouts/Layout.js"); // example: var data = requireScript("/data.xml?id=" + id, true); // // // -------------------------------------------------------------------- // // require.script( <javaScriptFileName>[, callback ] ); // Returns script execution result. // Script will be executed only once in every window where this method is called. // Script sources will be searched in parent windows and cached in current window. // Dynamic scripts are not cached ever, use them for data access, be sure not to // initialize global variables in you scripts. // example: requireScript("Layouts/Layout.js", function(Layout){}); // // -------------------------------------------------------------------- // // requireSource( <javaScriptFileName>[, dynamic ] ); // Returns script source. // example: var data = requireSource("data/index.jsld"); // // -------------------------------------------------------------------- // // // require.source( <javaScriptFileName>[, callback ] ); // Returns script source. // example: requireSource("data/index.jsld", function(str){}); // // -------------------------------------------------------------------- // // require.style( <cssFileName>[, callback ] ); // Applies style to this window. // Style will be applied only once in every window where this method is called. // Style sources will be searched in parent windows and cached in current window. // example: require.style("controls/menu/menu.css"); // example: require.style("controls/menu/menu.css", function(x){x.destroy()}); // // Resulting object will have following properties: // name, // value, name // style, // value, actual style element // destroy, // method, destroys style object, removes stylesheet from document // release, // method, when last reference released destroy() called automatically // // In case of failure the callback argument is null. // -------------------------------------------------------------------- // // // // will search scripts in own folder + "/js", will search styles in own folder + "/css". // if require's path ends with "/js/" than 'own' path will be one folder up.
